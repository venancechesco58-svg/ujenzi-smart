Firestore Security Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection - public read, write own
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth.uid == userId;
    }

    // Admins collection - admin only
    match /admins/{adminId} {
      allow read: if request.auth != null && exists(/databases/$(database)/documents/admins/$(request.auth.uid));
      allow write: if request.auth != null && get(/databases/$(database)/documents/admins/$(request.auth.uid)).data.adminLevel >= 2;
    }

    // Suppliers collection - supplier write own, public read
    match /suppliers/{supplierId} {
      allow read: if request.auth != null;
      allow write: if request.auth.uid == supplierId;
    }
  }
}
